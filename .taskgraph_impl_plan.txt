### 2. `scripts/healthcheck.js`

```js
// scripts/healthcheck.js
// Zero-dependency repo healthcheck. Node built-ins only. CommonJS. No shebang.

const fs = require('fs');
const path = require('path');

// Resolve repo root from THIS file's location, never process.cwd().
const ROOT = path.resolve(__dirname, '..');

// Single manifest of required files. Deliberately excludes .gitignore.
const REQUIRED_FILES = ['package.json', 'README.md', 'scripts/healthcheck.js'];

// Script keys that must be present in package.json (keys only, not values).
const REQUIRED_SCRIPT_KEYS = ['healthcheck', 'test'];

const failures = [];

// 1. File existence checks.
for (const rel of REQUIRED_FILES) {
  const abs = path.resolve(ROOT, rel);
  if (fs.existsSync(abs)) {
    console.log(`ok    file present: ${rel}`);
  } else {
    console.log(`fail  missing required file: ${rel}`);
    failures.push(`missing required file: ${rel}`);
  }
}

// 2. Parse and validate package.json manifest.
const pkgPath = path.resolve(ROOT, 'package.json');
let pkg = null;

if (fs.existsSync(pkgPath)) {
  let raw;
  try {
    raw = fs.readFileSync(pkgPath, 'utf8');
    pkg = JSON.parse(raw);
    console.log('ok    package.json is valid JSON');
  } catch (err) {
    console.log(`fail  package.json is invalid JSON: ${err.message}`);
    failures.push(`invalid JSON in package.json: ${err.message}`);
  }
}

// 3. Verify required script keys exist (presence only).
if (pkg) {
  const scripts = pkg.scripts || {};
  for (const key of REQUIRED_SCRIPT_KEYS) {
    if (Object.prototype.hasOwnProperty.call(scripts, key)) {
      console.log(`ok    scripts.${key} key present`);
    } else {
      console.log(`fail  missing required script key: scripts.${key}`);
      failures.push(`missing required script key: scripts.${key}`);
    }
  }
}

// 4. Collect-all, then exit.
if (failures.length > 0) {
  console.log(`\nhealthcheck FAILED with ${failures.length} problem(s):`);
  for (const f of failures) console.log(`  - ${f}`);
  process.exit(1);
}

console.log('\nhealthcheck OK');
process.exit(0);
```

Design decisions tied to constraints:
- Only `fs`, `path` required; `process` is a global. No external `require`.
- `path.resolve(__dirname, '..')` → root, independent of cwd.
- `JSON.parse` wrapped in try/catch → invalid JSON becomes a clean exit-1 failure, not a crash.
- Script validation checks **key presence** via `hasOwnProperty`, never literal command values.
- All failures accumulate in `failures[]`; single exit decision at the end.
- No shebang line.

### 3. `README.md`

Must be non-empty and cover **purpose**, **usage**, and **how the healthcheck works**. Skeleton:

```markdown
# swarm-sandbox

## Purpose
A minimal, zero-dependency Node.js project that self-validates its own
repository structure and manifest integrity via a healthcheck script.

## Usage
Requires Node.js >= 18. No install step is needed.

    npm test              # runs the healthcheck
    npm run healthcheck   # same as above
    node scripts/healthcheck.js

Exit code 0 means healthy; exit code 1 means one or more checks failed.

## How the healthcheck works
`scripts/healthcheck.js` uses only Node built-in modules (fs, path, process).
It resolves the repo root from its own location (`__dirname/..`), then:
1. Confirms each file in the REQUIRED_FILES manifest exists.
2. Parses package.json (invalid JSON is reported as a failure, not a crash).
3. Confirms the `scripts.healthcheck` and `scripts.test` keys exist.
All failures are collected and printed before exiting 0 (pass) or 1 (fail).
```

## Negative Verification Procedure (AC-6)

Run each case, capture exit code and message, then restore state before the next.

**Case A — missing required file:**
```
mv README.md README.tmp
npm test; echo exit=$?          # expect exit=1, message "missing required file: README.md"
mv README.tmp README.md
```

**Case B — missing required script key:**
```
cp package.json package.json.bak
node -e "const fs=require('fs');const p=JSON.parse(fs.readFileSync('package.json','utf8'));delete p.scripts.healthcheck;fs.writeFileSync('package.json',JSON.stringify(p,null,2)+'\n')"
node scripts/healthcheck.js; echo exit=$?   # expect exit=1, "missing required script key: scripts.healthcheck"
mv package.json.bak package.json
```

**Case C — corrupt package.json:**
```
cp package.json package.json.bak
printf '{ this is not json' > package.json
node scripts/healthcheck.js; echo exit=$?   # expect exit=1, "invalid JSON in package.json: ..."
mv package.json.bak package.json
```

Note: use `node scripts/healthcheck.js` directly for Cases B/C (npm may print its own error wrapper), and confirm the healthcheck's own message appears in output.
